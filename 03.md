```bash
# 상위 디렉토리를 포함하여 디렉토리 생성
mkdir -p /project/test
# -p : 부모 디렉토리(상위 디렉토리) 미 존재시 자동 생성
# /project 디렉토리 미 존재 상태에서 -p 옵션 없이 mkdir /project/test 실행 시 오류 발생

# /home/user1 디렉토리에 hello.txt 파일을 생성 후 Hello Rocky! 문자열 기록
cat >> /home/user1/hello.txt
Hello Rocky!
^D
# /home/user/hello.txt 읽기
cat /home/user/hello.txt
Hello Rocky!

# vi 편집기로 실행
vi /home/user/hello.txt

# vi 편집기 사용 단축키
i : 입력
Esc : 입력 종료
:w : 저장
:q : 종료
:x : 저장 후 종료
:q! : 저장 없이 종료
```
